# Using popularity based recommender model to make predictions
def recommend(song_id):     
    song_id_recommendations = popularity_recommendations 
    song_id_recommendations["song_Id"] = song_id 
    cols = song_id_recommendations.columns.tolist() 
    cols = cols[-1:] + cols[:-1] 
    song_id_recommendations = song_id_recommendations[cols] 
          
    return song_id_recommendations
